export default class HelloWorld extends Component {
    state = {
      count: 0,
      obj: {
        a: 0,
        others: {}
      }
    }
    handleClick = () => {
      // 相同：异步更新
      // 不同：1. react新值覆盖旧值 2. react动态加属性
      this.setState({ count: 1 })
      this.setState({
        obj: {...this.state.obj, a: 1, b: 1},
      })
    }

    render() {
        return (
            <div onClick={this.handleClick}>
          		{this.state.count}
                {this.state.obj.a}
				<Hello />
			<div>
        )
    }
}
